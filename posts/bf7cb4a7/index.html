void ExMDI::loadFromFile(QString &fileName)
{
    QFile file(fileName);  //ж–Үд»¶д»ҘиҜ»зҡ„ж–№ејҸиҜ»еҮә

    if (file.open(QIODevice::ReadOnly | QIODevice::Text)) {
        QTextStream stream(&file);  //д»Ҙж–Үжң¬жөҒж–№ејҸиҜ»еҸ–ж–Үд»¶
        ui->plainTextEdit->clear();
        ui->plainTextEdit->setPlainText(stream.readAll());
        file.close();

        m_currentFile = fileName;
        QFileInfo fileInfo(fileName);    //ж–Үд»¶дҝЎжҒҜ
        setWindowTitle(fileInfo.fileName());
        m_fileOpened = true;
    }
}

void ExMainWindow::on_actOpen_triggered()
{
    bool needNew = false;

    ExMDI *mdi = nullptr;

    if (ui->mdiArea->subWindowList().count() > 0) {  //еҰӮжһңжңүжү“ејҖзҡ„дё»зӘ—еҸЈпјҢиҺ·еҸ–жҙ»еҠЁзӘ—еҸЈ
        mdi =  (ExMDI *)ui->mdiArea->activeSubWindow()->widget();
        needNew = mdi->isFileOpended();   //ж–Үд»¶е·Із»Ҹжү“ејҖпјҢйңҖиҰҒж–°е»әзӘ—еҸЈ
    } else {
        needNew = true;
    }

    QString curPath = QDir::currentPath();
    QString fileName = QFileDialog::getOpenFileName(this, "жү“ејҖдёҖдёӘж–Үд»¶", curPath, "CзЁӢеәҸж–Үд»¶(*.h *cpp);;ж–Үжң¬ж–Үд»¶(*.txt);;жүҖжңүж–Үд»¶(*.*)");

    if (fileName.isEmpty())
        return;

    if (needNew) {
        mdi = new ExMDI(this);
        ui->mdiArea->addSubWindow(mdi);
    }

    mdi->loadFromFile(fileName);
    mdi->show();

    ui->actCopy->setEnabled(true);
    ui->actCut->setEnabled(true);
    ui->actPaste->setEnabled(true);
    ui->actFont->setEnabled(true);
}
